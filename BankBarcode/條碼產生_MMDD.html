function generateQRCode() { // 取得條碼輸入值 var firstBarcode = document.getElementById("firstBarcode").value; var secondBarcode = document.getElementById("secondBarcode").value; var paymentDue = document.getElementById("paymentDue").value; var paymentAmount = document.getElementById("paymentAmount").value; // 轉換第三段條碼（與原本相同邏輯） var convertedFirstBarcode = convertLettersToNumbers(firstBarcode); var convertedSecondBarcode = convertLettersToNumbers(secondBarcode); var oddSum = calculateOddSum(convertedFirstBarcode) + calculateOddSum(convertedSecondBarcode) + calculateOddSum(paymentDue.substring(3) + paymentAmount.padStart(9, '0')); var evenSum = calculateEvenSum(convertedFirstBarcode) + calculateEvenSum(convertedSecondBarcode) + calculateEvenSum(paymentDue.substring(3) + paymentAmount.padStart(9, '0')); var oddChecksum = oddSum % 11 === 0 ? 'A' : oddSum % 11 === 10 ? 'B' : oddSum % 11; var evenChecksum = evenSum % 11 === 0 ? 'X' : evenSum % 11 === 10 ? 'Y' : evenSum % 11; var thirdBarcode = paymentDue.substring(3) + oddChecksum + evenChecksum + paymentAmount.padStart(9, '0'); // 組成 QRCode 內容 var content = `FAMI||{"bar1":"${firstBarcode}","bar2":"${secondBarcode}","bar3":"${thirdBarcode}","ID":"3A01"}`; // 顯示 QRCode var qrcodeContainer = document.getElementById("qrcode"); qrcodeContainer.innerHTML = ""; // 清空舊的 QRCode new QRCode(qrcodeContainer, { text: content, width: 256, height: 256 }); }
